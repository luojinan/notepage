function compose(fn) {
  const arr = [fn] // 暂存函数
  return function temp(fnOrArg) {
    if(typeof fnOrArg === 'function') {
      arr.push(fnOrArg)
      return temp
    } else {
      // 执行函数
      const res = arr.reduce((val, fn)=>{
        return fn(val)
      }, fnOrArg)
      return res
    }
  }
}
