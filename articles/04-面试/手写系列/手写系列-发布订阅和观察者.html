class Observer {
  constructor() {
    this.message = {} // ж¶ҲжҒҜйҳҹеҲ—
  }

  $on(type, callback) {
    // еҰӮжһңжІЎжңүиҝҷдёӘеұһжҖ§пјҢе°ұеҲқе§ӢеҢ–дёҖдёӘз©әзҡ„ж•°з»„
    if (!this.message[type]) {
      this.message[type] = [];
    }
    this.message[type].push(callback);
  }

  $off(type, callback) {
    if (!this.message[type]) return;
    // еҲӨж–ӯжҳҜеҗҰжңүcallbackиҝҷдёӘеҸӮж•°
    if (!callback) {
      this.message[type] = undefined;
      return;
    }
    // еҰӮжһңжңүcallback,е°ұд»…д»…еҲ жҺүcallbackиҝҷдёӘж¶ҲжҒҜ(иҝҮж»ӨжҺүиҝҷдёӘж¶ҲжҒҜж–№жі•)
    this.message[type] = this.message[type].filter((item) => item !== callback);
  }
  $emit(type) {
    if(!this.message[type]) return;
    // жҢЁдёӘжү§иЎҢжҜҸдёҖдёӘж¶ҲжҒҜзҡ„еӣһи°ғеҮҪж•°callback
    this.message[type].forEach(item => {
      item()
    });
  }
}

  constructor(cb){
    if (typeof cb === 'function') {

    } else {

    }
  }

  }

class Subject {
  constructor() {
    // з»ҙжҠӨи§ӮеҜҹиҖ…еҲ—иЎЁ
    this.observerList = []
  }
  addObserver(observer) {
    this.observerList.push(observer)
  }
  notify() {
    this.observerList.forEach(observer => {
      observer.update() // еҸ‘еёғйҒҚеҺҶи§ҰеҸ‘зҡ„жҳҜе®һдҫӢйҮҢзҡ„update
    })
  }
