дј йҖ’иҝӣ Promise зҡ„еӣһи°ғеҮҪж•°зҡ„еҸӮж•°з”ұ Promise жҸҗдҫӣ

js
// Promise еҶ…йғЁзҡ„3дёӘзҠ¶жҖҒ
const PENDING = 'pending' // зӯүеҫ…зҠ¶жҖҒ
const RESOLVED = 'resolved' // е®ҢжҲҗзҠ¶жҖҒ
const REJECTED = 'rejected' // жӢ’з»қзҠ¶жҖҒ


function MyPromise(fn) {
  this.state = PENDING
  this.res = null

  // Promise жҸҗдҫӣеӣһи°ғеҮҪж•°зҡ„2дёӘеҸӮж•° resolveеҮҪж•° rejectеҮҪж•°
  function resolve(value) {
    // еҸӘд»Һзӯүеҫ…зҠ¶жҖҒ дҝ®ж”№дёә е®ҢжҲҗпјҢе·Із»Ҹдҝ®ж”№иҝҮзҠ¶жҖҒзҡ„дёҚдјҡжү§иЎҢ
    if (this.state === PENDING) {
      this.state = RESOLVED // дҝ®ж”№зҠ¶жҖҒдёә е®ҢжҲҗ
      this.res = value // и®°еҪ•res
    }
  }
  function reject(value) {
    if (this.state === PENDING) {
      this.state = REJECTED // дҝ®ж”№зҠ¶жҖҒдёә жӢ’з»қ
      this.res = value // и®°еҪ•res
    }
  }
}
MyPromise.prototype.then = function (onFulfilled, onRejected){}

е®ҢжҲҗжҲ–иҖ…жӢ’з»қзҠ¶жҖҒи§ҰеҸ‘ж—¶пјҢжүҫеҲ°thenдёӯзӣёеә”зҡ„еӣһи°ғеҮҪж•°жү§иЎҢ еӣ жӯӨthenеҮҪж•°пјҢе…¶е®һжҳҜжҠҠеӣһи°ғеҮҪж•°еӯҳиө·жқҘпјҢеҲ°ж—¶жңәдәҶз”ұдҝ®ж”№зҠ¶жҖҒж—¶и§ҰеҸ‘

MyPromise.prototype.then = function (onFulfilled, onRejected){

  this.onRejected = onRejected

еңЁresolveеҮҪж•°е’ҢrejectеҮҪж•°и§ҰеҸ‘дҝ®ж”№зҠ¶жҖҒж—¶и§ҰеҸ‘ then дёӯеӯҳеӮЁзҡ„еӣһи°ғеҮҪж•°

js
// Promise еҶ…йғЁзҡ„3дёӘзҠ¶жҖҒ
const PENDING = 'pending' // зӯүеҫ…зҠ¶жҖҒ
const RESOLVED = 'resolved' // е®ҢжҲҗзҠ¶жҖҒ
const REJECTED = 'rejected' // жӢ’з»қзҠ¶жҖҒ


function MyPromise(fn) {
  this.state = PENDING
  this.res = null

  this.onFulfilled = null
  this.onRejected = null

  // Promise жҸҗдҫӣеӣһи°ғеҮҪж•°зҡ„2дёӘеҸӮж•° resolveеҮҪж•° rejectеҮҪж•°
  function resolve(value) {
    // еҸӘд»Һзӯүеҫ…зҠ¶жҖҒ дҝ®ж”№дёә е®ҢжҲҗпјҢе·Із»Ҹдҝ®ж”№иҝҮзҠ¶жҖҒзҡ„дёҚдјҡжү§иЎҢ
    if (this.state === PENDING) {
      this.state = RESOLVED // дҝ®ж”№зҠ¶жҖҒдёә е®ҢжҲҗ
      this.res = value // и®°еҪ•res
      this.onFulfilled()
    }
  }
  function reject(value) {
    if (this.state === PENDING) {
      this.state = REJECTED // дҝ®ж”№зҠ¶жҖҒдёә жӢ’з»қ
      this.res = value // и®°еҪ•res
      this.onRejected()
    }
  }
}
MyPromise.prototype.then = function (onFulfilled, onRejected){
  this.onFulfilled = onFulfilled
  this.onRejected = onRejected
}

js
// Promise еҶ…йғЁзҡ„3дёӘзҠ¶жҖҒ
const PENDING = 'pending' // зӯүеҫ…зҠ¶жҖҒ
const RESOLVED = 'resolved' // е®ҢжҲҗзҠ¶жҖҒ
const REJECTED = 'rejected' // жӢ’з»қзҠ¶жҖҒ

function MyPromise(fn) {
  this.state = PENDING
  this.res = null

  this.onFulfilled = null
  this.onRejected = null

  // Promise жҸҗдҫӣеӣһи°ғеҮҪж•°зҡ„2дёӘеҸӮж•° resolveеҮҪж•° rejectеҮҪж•°
  function resolve(value) {
    // еҸӘд»Һзӯүеҫ…зҠ¶жҖҒ дҝ®ж”№дёә е®ҢжҲҗпјҢе·Із»Ҹдҝ®ж”№иҝҮзҠ¶жҖҒзҡ„дёҚдјҡжү§иЎҢ
    if (this.state === PENDING) {
      this.state = RESOLVED // дҝ®ж”№зҠ¶жҖҒдёә е®ҢжҲҗ
      this.res = value // и®°еҪ•res
      this.onFulfilled(this.res)
    }
  }
  function reject(value) {
    if (this.state === PENDING) {
      this.state = REJECTED // дҝ®ж”№зҠ¶жҖҒдёә жӢ’з»қ
      this.res = value // и®°еҪ•res
      this.onRejected()
    }
  }
  
  // ејҖе§Ӣ
  fn(resolve.bind(this),reject.bind(this))
}
MyPromise.prototype.then = function (onFulfilled, onRejected){
  this.onFulfilled = onFulfilled
  this.onRejected = onRejected
}

// жөӢиҜ•д»Јз Ғ
new MyPromise((resolve, reject) => {
  setTimeout(() => {
    resolve(1)
  }, 0)
}).then(value => {
  console.log(value)
})
